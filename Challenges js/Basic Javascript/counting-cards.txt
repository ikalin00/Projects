let count = 0;

function cc(card) {

 
  // Only change code below this line
   let cards = card 
   if(cards<7 && card>1){
     count++
   }
   else if(cards === 10 || cards === "J" ||  cards === "Q" ||  cards === "K" ||  cards === "A" ){
     count--
   }
  
  if(count>0)
{
  return count + " Bet" 
}

else

  return count + " Hold";
  // Only change code above this line
}

cc(2); cc(3); cc(7); cc('K'); cc('A');
